66 void GoodParser::skipWhitespace() {
67 while (m_readPos < m_data.size()) {
68 char c = m_data[m_readPos];
69 if (c == '\n') {
70 ++m_lineIndex; // if char is newline, increment lineIndex
71 ++m_readPos; // then skip it
72 } else if (WHITESPACE.find(c) != std::string::npos) {
73 ++m_readPos; // if char is whitespace, skip it
74 } else {
75 break; // no more whitespace

260

262 skipWhitespace();

264 consume(1); // consume '}'
265 m_depth--;
266 // validate server for mandatory keys
267 thisServer.validate();
268 // if valid, push to server configs
