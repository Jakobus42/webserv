70 void GoodParser::skipWhitespace() {

72 char c = m_data[m_readPos];

240 void GoodParser::expectServerBlock() throw(parse_exception) {
241 if (!matchToken("server")) {
242 throw parse_exception(m_lineIndex, "Expected 'server' keyword");
243 }
244 skipWhitespace();
245 if (m_readPos >= m_data.size() || m_data[m_readPos] != '{') {
246 throw parse_exception(m_lineIndex, "Expected opening brace after Server");
247 }
248 consume(1); // consume '{'
249 m_depth++;
250
251 Server thisServer;
252
253 while (m_readPos < m_data.size()) {
254 skipWhitespace();
255 if (m_readPos < m_data.size() && m_data[m_readPos] == '{') {
256 throw parse_exception(m_lineIndex, "Unexpected opening brace in Server block");
257 }
258 if (m_readPos < m_data.size() && m_data[m_readPos] == '}') {
259 consume(1); // consume '}'
260 m_depth--;
261 thisServer.validate();
262 m_configs.push_back(thisServer);
263 return;
264 }
265 std::string token = readToken();
266 if (matchToken("location")) {
267 expectLocationBlock(thisServer.location);
268 } else if (!token.empty()) {
269 CommandType type = matchDirective(token, serverDirectives());
270 if (type == _D_NOT_VALID) {
271 throw parse_exception(m_lineIndex, "Unexpected directive in Server block: " + token);
272 }
273 std::string value = readValue();
274 setServerValue(value, type, thisServer);
275 }
276 }
277 throw parse_exception(m_lineIndex, "Server block not closed with '}'");
278 }

294 void GoodParser::expectLocationBlock(Location& parentLocation) throw(parse_exception) {
295 std::string path = readToken();
296
297 if (path.empty()) {
298 throw parse_exception(m_lineIndex, "Location has no path");
299 }
300 if (!isValidPath(path)) {
301 throw parse_exception(m_lineIndex, "Path is invalid");
302 }
303 consume(path.length());
304 skipWhitespace();
305 if (m_readPos >= m_data.size() || m_data[m_readPos] != '{') {
306 throw parse_exception(m_lineIndex, "Expected '{' after location path");
307 }
308 consume(1); // consume '{'
309 m_depth++;
310
311 Location thisLocation;
312
313 thisLocation.path = http::Router::splitPath(path);
314 while (m_readPos <= m_data.size()) {
315 skipWhitespace();
316 if (m_readPos < m_data.size() && m_data[m_readPos] == '{') {
317 throw parse_exception(m_lineIndex, "Unexpected opening brace in location block");
318 }
319 if (m_readPos < m_data.size() && m_data[m_readPos] == '}') {
320 consume(1); // consume '}'
321 m_depth--;
322 thisLocation.validate();
323 // validate Location whether enough stuff is present
324 // is anything mandatory for a location?
325 parentLocation.locations.push_back(thisLocation);
326 return;
327 }
328 std::string token = readToken();
329 if (matchToken("location")) {
330 expectLocationBlock(thisLocation);
331 } else if (!token.empty()) {
332 // process location directives
333 CommandType type = matchDirective(token, locationDirectives());

435 void GoodParser::parseLimitExcept(const std::string& value, Location& location) {
436 static std::map<std::string, http::Method> implementedMethods;
437
438 if (implementedMethods.empty()) {
439 implementedMethods["GET"] = http::GET;
440 implementedMethods["POST"] = http::POST;
441 implementedMethods["DELETE"] = http::DELETE;
442 }
443
444 std::set<http::Method> allowedMethods;
445 std::stringstream ss(value);
446 std::string token;
447
448 while (ss >> token) {
449 if (implementedMethods.find(token) == implementedMethods.end()) {
450 throw parse_exception(m_lineIndex, "Unexpected method for limit_except: " + token);
451 }
452 if (allowedMethods.find(implementedMethods[token]) != allowedMethods.end()) {
453 throw parse_exception(m_lineIndex, "Duplicte method for limit_except: " + token);
454 }
455 allowedMethods.insert(implementedMethods[token]);
456 }
457 if (allowedMethods.empty()) {
458 throw parse_exception(m_lineIndex, "Expected value for limit_except");
459 }
460 location.allowedMethods = allowedMethods;
