202 bool Response::parseAbsoluteForm(const std::string &path, const Request& request, t_PathData &pathData)
203 {
204 (void)request;
205 std::string scheme;
206 std::string query = "";
207 std::string authority;
208 std::string pure_path;
209 if (path.find(':') == std::string::npos) {
210 return false;
211 }
212 scheme = path.substr(0, path.find(":")); // scheme before the first colon
213 if (scheme == "" || std::isalpha(scheme[0]) == 0
214 || scheme.find_first_not_of("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789+-.") != std::string::npos) {
215 return false;
216 }
217 if (path.find('?') != std::string::npos) { // optional query part after the first question mark
218 query = path.substr(path.find('?') +1);
219 }
220 pure_path = path.substr(path.find(':') +1, path.find('?') - path.find(':') -1);
221 if (pure_path[0] != '/' && pure_path[1] != '/') { // authority part starts with two slashes
222 return false;
223
224 }
225 pure_path = pure_path.substr(2);
226 if (pure_path.find('/') == std::string::npos) {
227 return false;
228 }
229 authority = pure_path.substr(0, pure_path.find('/'));
230 if (authority.empty()) {
231 return false;
232 }
233 pure_path = pure_path.substr(pure_path.find('/'));
234 pathData.query = query;
235 pathData.pure_path = pure_path;
236 pathData.scheme = scheme;
237 pathData.authority = authority;
238 return true;
239 }

241 bool Response::parseOriginForm(const std::string &path, const Request& request, t_PathData &pathData)
242 {
243 std::string scheme;
244 std::string authority;
245 std::string pure_path;
246 std::string query;
247 if (path.find('?') != std::string::npos) { // optional query part after the first question mark
248 query = path.substr(path.find('?') +1);

250 pure_path = path.substr(0, path.find('?'));
251 if (pure_path[0] != '/') {
252 return false;
