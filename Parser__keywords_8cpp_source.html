12 int ft_stoi(std::string str) {
13 long res = 0;
14 unsigned long i = 0;
15 if (str.size() == 0) {
16 return -1;
17 }
18 for (; i < str.size(); i++) {
19 if (!std::isdigit(str[i]))
20 return -1;
21 res = res * 10 + str[i] - '0';
22 if (res > 2147483647)
23 return -1;
24 }
25 return res;
