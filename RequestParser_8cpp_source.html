144 // look for '?', if found take the string afterwards and take from it key=value pairs
145 // separated by '&', put each in the map, if a duplicate key is found simply overwrite its value
146 void RequestParser::parseQuery() {
147 std::map<std::string, std::string>& queryParameters = m_req.getUri().query;
148 const std::string& uriRaw = m_req.getUriRaw();
149 std::size_t currentPos = uriRaw.find_first_of('?');
150 std::size_t end = uriRaw.find_first_of('#');
151
152 if (currentPos == std::string::npos) {
153 return;
154 }
155
156 if (uriRaw[currentPos] == '?') {
157 currentPos++;
158 }
159
160 // abc?a=b&cde=fgh&jk=l&mnop=qrst
161 while (currentPos <= end) {
162 std::string currentKey = "";
163 std::string currentValue = "";
164
165 std::size_t nextDelimiter = uriRaw.find_first_of('=', currentPos);
166 currentKey = uriRaw.substr(currentPos, nextDelimiter - currentPos);
167 std::size_t nextAmpersand = uriRaw.find_first_of('&', currentPos);
168 if (nextAmpersand == std::string::npos && nextDelimiter == std::string::npos) {
169 // fuck this
170 break;
171 }
172 if (nextAmpersand <= nextDelimiter) {
173 // no equal sign
174 // ...&key&key&key=value
175 // or ...&key&key&key
176 // should advance to the next one
177 currentPos = nextAmpersand + 1;
178 continue;
179 }
180
181 if (nextDelimiter >= end) {
182 // no equal sign, no value gets inserted
183 // ...&key=value&key
184 // or ...&key=value&key#fragment
185 break;
186 }
187 currentPos = nextDelimiter + 1;
188 if (currentPos >= end) {
189 // no value but equal sign, empty value gets inserted
190 // ...&key=value&key=
191 // or ...&key=value&key=#fragment
192 // this should just work
193 }
194 nextDelimiter = uriRaw.find_first_of('&', currentPos);
195 if (nextDelimiter >= end) {
196 // last parameter, delimited by a fragment
197 // ..&key=value#fragment
198 nextDelimiter = end;
199 }
200 currentValue = uriRaw.substr(currentPos, nextDelimiter - currentPos);
201 if (!currentKey.empty() && queryParameters.find(currentKey) == queryParameters.end()) {
202 queryParameters.insert(std::make_pair(currentKey, currentValue));
203 }
204 if (nextDelimiter == end) {
205 // reached end, exit loop
206 break;
207 }
208 currentPos = nextDelimiter + 1;
209 }

212 void RequestParser::parseHeaders() {
213 while (true) {
214 char* line = this->readLine();
215 if (!line) {
216 return;
217 } else if (!*line) {
218 break;
219 }
220
221 Token key = this->extractHeaderKey(line);
222 if (!*line) {
223 m_req.setHeader(key.token, key.size, NULL, 0);
224 }
225 std::size_t valCount = 0;
226 while (*line) {
227 Token value = this->extractHeaderValue(line);
228 m_req.setHeader(key.token, key.size, value.token, value.size);
229 if (++valCount > MAX_HEADER_VALUE_COUNT) {
230 throw http::exception(PAYLOAD_TOO_LARGE, "header value count too large");
231 }
232 }
233 if (m_req.getHeaders().size() > MAX_HEADER_COUNT) {
234 throw http::exception(PAYLOAD_TOO_LARGE, "header count too large");
235 }
236 }
237 if (m_state == HEADERS) {
238 this->interpretHeaders();
239 } else if (m_state == TRAILING_HEADERS) {
240 this->setState(COMPLETE);
241 }

244 RequestParser::Token RequestParser::extractHeaderKey(char*& line) {
245 std::size_t keySize = 0;
246 for (; *line && *line != ':'; ++line) {
247 if (this->isTChar(*line) != true) {
248 throw http::exception(BAD_REQUEST, "malformed header name: expected tchar");
249 }
250 *line = std::tolower(*line);
251 keySize++;
252 }
253
254 if (keySize == 0) {
255 throw http::exception(BAD_REQUEST, "malformed header name: expected size > 0");
256 } else if (*line != ':') {
257 throw http::exception(BAD_REQUEST, "malformed header name: missing colon separator");
258 } else if (std::strchr(WHITESPACE, *(line - 1))) {
259 throw http::exception(BAD_REQUEST, "malformed header name: no whitespace allowed before colon");
260 }
261
262 if (keySize > MAX_HEADER_NAME_LENGTH) {
263 throw http::exception(PAYLOAD_TOO_LARGE, "header key too large");
264 }
265
266 return Token(line++ - keySize, keySize);
267 }

270 RequestParser::Token RequestParser::extractHeaderValue(char*& line) {
271 const char* value = line + std::strspn(line, WHITESPACE);
272 std::size_t raw_size = std::strcspn(value, ",");
273
274 std::size_t size = raw_size;
275 while (size > 0 && std::strchr(WHITESPACE, value[size - 1])) {
276 --size;
277 }
278
279 line = const_cast<char*>(value + raw_size);
280 if (*line == ',') {
281 ++line;
282 }
283
284 if (size > MAX_HEADER_VALUE_LENGTH) {
285 throw http::exception(PAYLOAD_TOO_LARGE, "header value length too large");
286 }
287 return Token(value, size);
