223 void RequestParser::parseHeaders() {
224 while (true) {
225 char* line = this->readLine();
226 if (!line) {
227 return;
228 } else if (!*line) {
229 break;
230 }
231
232 Token key = this->extractHeaderKey(line);

235 }
236 std::size_t valCount = 0;

242 }

246 }
247 }
248 if (m_state == HEADERS) {
249 this->interpretHeaders();
250 } else if (m_state == TRAILING_HEADERS) {
251 this->setState(COMPLETE);

255 RequestParser::Token RequestParser::extractHeaderKey(char*& line) {
256 std::size_t keySize = 0;
257 for (; *line && *line != ':'; ++line) {
258 if (this->isTChar(*line) != true) {
259 throw http::exception(BAD_REQUEST, "malformed header name: expected tchar");
260 }
261 *line = std::tolower(*line);
262 keySize++;
263 }
264
265 if (keySize == 0) {
266 throw http::exception(BAD_REQUEST, "malformed header name: expected size > 0");
267 } else if (*line != ':') {
268 throw http::exception(BAD_REQUEST, "malformed header name: missing colon separator");
269 } else if (std::strchr(WHITESPACE, *(line - 1))) {
270 throw http::exception(BAD_REQUEST, "malformed header name: no whitespace allowed before colon");
271 }
272
273 if (keySize > MAX_HEADER_NAME_LENGTH) {
274 throw http::exception(PAYLOAD_TOO_LARGE, "header key too large");
275 }
276
277 return Token(line++ - keySize, keySize);

335 void RequestParser::parseData() {
336 if (m_contentLength) {
337 size_t available = std::min(m_buffer.size(), m_contentLength);
338 if (available == 0) {
339 this->setPendingState();

341 }
342
343 m_req.appendToBody(m_buffer.getReadPos(), available);
344 if (m_req.getBody().size() > MAX_BODY_SIZE) {
345 throw http::exception(PAYLOAD_TOO_LARGE, "body too large");
346 }
347 m_buffer.consume(available);

349 if (m_contentLength > 0) {
350 this->setPendingState();

352 }
353 }

356 this->setState(COMPLETE);
