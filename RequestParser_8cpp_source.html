73 void RequestParser::parseHeaders() {
74 while (true) {
75 char* line = this->readLine();
76 if (!line) {
77 return;
78 } else if (!*line) {
79 break;

85 }
86 std::size_t valCount = 0;

88 Token value = this->extractHeaderValue(line);
89 m_req.setHeader(key.token, key.size, value.token, value.size);

91 throw http::exception(PAYLOAD_TOO_LARGE, "header value count too large");
92 }
93 }

95 throw http::exception(PAYLOAD_TOO_LARGE, "header count too large");
96 }
97 }
98 if (m_state == HEADERS) {
99 this->interpretHeaders();
100 } else if (m_state == TRAILING_HEADERS) {

105 RequestParser::Token RequestParser::extractHeaderKey(char*& line) {
106 std::size_t keySize = 0;
107 for (; *line && *line != ':'; ++line) {
108 if (this->isTChar(*line) != true) {
109 throw http::exception(BAD_REQUEST, "malformed header name: expected tchar");
110 }
111 *line = std::tolower(*line);
112 keySize++;
113 }
114
115 if (keySize == 0) {
116 throw http::exception(BAD_REQUEST, "malformed header name: expected size > 0");
117 } else if (*line != ':') {
118 throw http::exception(BAD_REQUEST, "malformed header name: missing colon separator");
119 } else if (std::strchr(WHITESPACE, *(line - 1))) {
120 throw http::exception(BAD_REQUEST, "malformed header name: no whitespace allowed before colon");
121 }
122
123 if (keySize > MAX_HEADER_NAME_LENGTH) {
124 throw http::exception(PAYLOAD_TOO_LARGE, "header key too large");
125 }
126
127 return Token(line++ - keySize, keySize);
128 }

167 void RequestParser::parseData() {
168 if (m_contentLength) {
169 size_t available = std::min(m_buffer.size(), m_contentLength);
170 if (available == 0) {
171 this->setPendingState();
172 return;
173 }
174
175 m_req.appendToBody(m_buffer.getReadPos(), available);
176 if (m_req.getBody().size() > MAX_BODY_SIZE) {
177 throw http::exception(PAYLOAD_TOO_LARGE, "body too large");
178 }
179 m_buffer.consume(available);
180 m_contentLength -= available;
181 if (m_contentLength > 0) {
182 this->setPendingState();
183 return;
184 }
185 }
186
187
188 if (m_state == BODY) {
189 this->setState(COMPLETE);
190 } else if (m_state == CHUNK_DATA) {
191 if (m_buffer.size() < CRLF.length()) {
192 this->setPendingState();
193 } else {
194 m_buffer.consume(CRLF.length());
195 this->setState(CHUNK_SIZE);
