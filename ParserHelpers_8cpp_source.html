13 int Parser::handlePrompt(std::string& line, int layer, const int& lineCount) {
14 // return 0 if successful and semicolon is found, 1 if semicolon is not
15 // found, 2 if error cut off trailing and starting spaces
16 int qouteFlag = 1;
17 while (line[line.size() - 1] == ' ') {
18 line.erase(line.size() - 1);
19 }
20 while (line[0] == ' ') {
21 line.erase(0, 1);
22 }
23 if (line[line.size() - 1] == ';') {
24 qouteFlag = 0;
25 line.erase(line.size() - 1);
26 }
27 if (line.find(';') != std::string::npos) {
28 std::cout << "Configuration file (line " << lineCount << "): "
29 << "Missplaced semicolon (;) found"
30 << std::endl;
31 return 2;
32 }
33 std::vector<std::string> args;
34 std::stringstream stream(line);
35 std::string key;
36 while (getline(stream, key, ' ')) {
37 args.push_back(key);
38 }
39 if (saveConfigData(args, layer, qouteFlag, lineCount) == 1) {
40 return 2;
41 }
42 return qouteFlag;
43 }

233 int Parser::saveConfigData(std::vector<std::string>& args, int layer, int qoute_flag, const int& lineCount) {
234 enum CmdId command_id = idCommand(args[0]);
235 if (command_id == UNKNOWN_ID) {
236 return unknownCommandError(lineCount);
237 }
238 if ((layer == 0 && qoute_flag == 0) || (layer == 0 && command_id != SERVER_ID) || (layer > 0 && qoute_flag != 0 && command_id != LOCATION_ID)) {
239 return unexpectedCommandError(lineCount);
240 }
241
242 if (layer == 0) { // global
243 return server(args, lineCount, layer);
244 }
245 if (layer == 1) { // server
246 if (qoute_flag != 0) {
247 return location(args, lineCount, layer);
248 }
249 switch (command_id) {
250 case LISTEN_ID:
251 return listen(args, lineCount, layer);
252 case SERVER_NAME_ID:
253 return serverName(args, lineCount, layer);
254 case ERROR_PAGE_ID:
255 return errorPage(args, lineCount, layer);
256 case CLIENT_MAX_BODY_SIZE_ID:
257 return clientMaxBodySize(args, lineCount, layer);
258 case GLOBAL_ROOT_ID:
259 return globalRoot(args, lineCount, layer);
260 case DATA_DIR_ID:
261 return dataDir(args, lineCount, layer);
262 case UPLOAD_DIR_ID:
263 return uploadDir(args, lineCount, layer);
264 case INDEX_ID:
265 return index(args, lineCount, layer);
266 default:
267 return unexpectedCommandError(lineCount);
268 }
269 }
270 if (layer > 1) { // location
271 if (qoute_flag != 0) {
272 return location(args, lineCount, layer);
273 }
274 switch (command_id) {
275 case LIMIT_EXCEPT_ID:
276 return limitExcept(args, lineCount, layer);
277 case RETURN_ID:
278 return returnKeyword(args, lineCount, layer);
279 case ROOT_ID:
280 return root(args, lineCount, layer);
281 case AUTOINDEX_ID:
282 return autoindex(args, lineCount, layer);
283 case UPLOAD_DIR_ID:
284 return uploadDir(args, lineCount, layer);
285 case INDEX_ID:
286 return index(args, lineCount, layer);
287 default:
288 return unexpectedCommandError(lineCount);
289 }
290 }
291 return 0;
