243 std::vector<std::string> file;
244 std::vector<std::string> result;
245 std::string path = "";
246 // Split into file and path if file is present
247 if (uri[uri.size() - 1] != '/') {

262 bool deeper = false;

267 if (comparePaths(it->path, result, len) == 0) {

275 return 1;
