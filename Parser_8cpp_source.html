303 void Parser::expectServerBlock() throw(parse_exception) {
304 skipWhitespace();
305 if (m_readPos >= m_data.size() || m_data[m_readPos] != '{') {
306 throw parse_exception(m_lineIndex, "Expected opening brace after Server");
307 }
308 consume(1); // consume '{'
309 m_depth++;
310
311 Server thisServer;
312
313 while (m_readPos < m_data.size()) {
314 skipWhitespace();
315 if (m_readPos < m_data.size() && m_data[m_readPos] == '{') {
316 throw parse_exception(m_lineIndex, "Unexpected opening brace in Server block");
317 }
318 if (m_readPos < m_data.size() && m_data[m_readPos] == '}') {
319 consume(1); // consume '}'
320 m_depth--;
321 thisServer.validate();
322 m_configs.push_back(thisServer);
323 return;
324 }
325 std::string token = readToken();
326 if (matchToken("location")) {
327 expectLocationBlock(thisServer.location);
328 } else if (!token.empty()) {
329 CommandType type = matchDirective(token, serverDirectives());
330 if (type == _D_NOT_VALID) {
331 throw parse_exception(m_lineIndex, "Unexpected directive in Server block: " + token);
332 }
333 std::string value = readValue();
334 processValue(value, type, thisServer);
335 }
336 }
337 throw parse_exception(m_lineIndex, "Server block not closed with '}'");
338 }

350 void Parser::expectLocationBlock(Location& parentLocation) throw(parse_exception) {
351 std::string path = readToken();
352
353 if (path.empty()) {
354 throw parse_exception(m_lineIndex, "Location has no path");
355 }
356 if (!isValidPath(path)) {
357 throw parse_exception(m_lineIndex, "Path is invalid");
358 }
359 consume(path.length());
360 skipWhitespace();
361 if (m_readPos >= m_data.size() || m_data[m_readPos] != '{') {
362 throw parse_exception(m_lineIndex, "Expected '{' after location path");
363 }
364 consume(1); // consume '{'
365 m_depth++;
366
367 Location thisLocation;
368
369 thisLocation.path = path;
370 thisLocation.pathAsTokens = shared::string::splitPath(path);
371 while (m_readPos <= m_data.size()) {
372 skipWhitespace();

374 throw parse_exception(m_lineIndex, "Unexpected opening brace in location block");
375 }
376 if (m_readPos < m_data.size() && m_data[m_readPos] == '}') {
377 consume(1); // consume '}'

436 uint32_t Parser::parseIpAddress(const std::string& host) {
437 uint32_t ip = 0;
438 std::size_t start = 0;
439 std::size_t part = 0;
440
441 while (part < 4) {
442 std::size_t pos = host.find('.', start);
443 std::string segment;
444 if (pos == std::string::npos) {
445 segment = host.substr(start);
446 } else {
447 segment = host.substr(start, pos - start);
448 }
449 if (segment.empty()) {
450 throw parse_exception(m_lineIndex, "Invalid IP address format: empty segment in \"" + host + "\"");
451 }
452 for (std::size_t i = 0; i < segment.size(); ++i) {
453 if (!std::isdigit(static_cast<unsigned char>(segment[i]))) {
454 throw parse_exception(m_lineIndex, "Invalid IP address format: unexpected character '" + segment.substr(i, 1) + "' in \"" + host + "\"");
455 }
456 }
457 if (segment.size() > 3) {
458 throw parse_exception(m_lineIndex, "Invalid IP address format: segment too long in \"" + host + "\"");
459 }
460 int octet = shared::string::toNum<int>(segment, 10);
461 if (octet < 0 || octet > 255) {
462 throw parse_exception(m_lineIndex, "IP segment out of range in \"" + host + "\"");
463 }
464 ip = (ip << 8) | (octet & 0xFF);
465 part++;
466 if (pos == std::string::npos)
467 break;
468 start = pos + 1;
469 }
470
471 std::cout << "parts: " << part << "; ip: " << ip << std::endl;
472 if (part != 4) {
473 throw parse_exception(m_lineIndex, "Invalid IP address format: \"" + host + "\"");
474 }
475 return ip;
476 }

478 void Parser::parseListen(const std::string& value, Server& server) {
479 std::string host;
480 std::string port;

493 if (ss >> port && !port.empty()) {
494 throw parse_exception(m_lineIndex, "Invalid value for listen: " + value);
