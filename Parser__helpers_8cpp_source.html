12 int ConfigFileParser::handlePrompt(std::string& line, int layer, int& lineCount) {
13  // return 0 if successful and semicolon is found, 1 if semicolon is not found, 2 if error
14  // cut off trailing and starting spaces
15  int qouteFlag = 1;
16  while (line[line.size() - 1] == ' ') {
17  line.erase(line.size() - 1);
18  }
19  while (line[0] == ' ') {
20  line.erase(0, 1);
21  }
22  if (line[line.size() - 1] == ';') {
23  qouteFlag = 0;
24  line.erase(line.size() - 1);
25  }
26  if (line.find(';') != std::string::npos) {
27  LOG("Configuration file (line " << lineCount << "): "
28  << "Missplaced semicolon (;) found" << std::endl,
29  logger::ERROR)
30  return 2;
31  }
32  std::vector<std::string> args;
33  std::stringstream stream(line);
34  std::string key;
35  while (getline(stream, key, ' ')) {
36  args.push_back(key);
37  }
38  if (SaveConfigData(args, layer, qouteFlag, lineCount) == 1) {
39  return 2;
40  }
41  return qouteFlag;
42 }
