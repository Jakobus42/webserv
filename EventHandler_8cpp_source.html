149 void EventHandler::buildResponse(void) {
150 if (m_requests.empty() || m_state > WRITING)
151 return;
152 const http::Request& currentRequest = m_requests.front();
153 http::Response& currentResponse = m_connection.responseBuffer;
154 currentResponse.doMagicToCalculateStatusCode(currentRequest);
155
156 // set response type: CGI, error, or normal
157 // i.e. look for file and error if not found, otherwise build that
158 // or, if cgi header found, validate, run and respond accordingly
159
160 switch (currentResponse.getType()) {
161 case (http::IDK_NORMAL_I_GUESS):
162 currentResponse.buildFromRequest(currentRequest);
163 break;
164 case (http::CGI):
165 currentResponse.buildCGIResponse(currentRequest);
166 break;
167 case (http::ERROR):
168 currentResponse.buildErrorResponse(currentRequest);
169 break;
170 }
171 if (currentResponse.done()) {
172 m_responses.push(currentResponse);
173 m_requests.pop();
174 setState(PENDING_SEND);
175 currentResponse.reset();
176 }
177 std::cout << "m_requests now " << m_requests.size() << " long" << std::endl;
178 }
